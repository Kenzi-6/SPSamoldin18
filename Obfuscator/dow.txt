package com.company;import java.io.*;import java.net.URL;import java.nio.channels.Channels;import java.nio.channels.ReadableByteChannel;import javazoom.jl.decoder.JavaLayerException;import javazoom.jl.pl.Player;public class dow { private static final String IN_FILE_TXT = "src/demo/download/inFile.txt"; public static void main(String[] args) { try (BufferedReader te = new BufferedReader(new FileReader(IN_FILE_TXT))) { String st; int co = 1; try { while (( st = te.readLine()) != null ) { String[] ma = st.split(" "); for (int i = 1; i < 2; i++){ String Ur = ma[0]; String Pa = ma[1]; do(Ur, Pa); System.out.println("Скачивание " + co +" файла завершено"); co ++; } } } catch (IOException e) { e.printStackTrace(); } } catch (IOException e) { e.printStackTrace(); } try (FileInputStream in = new FileInputStream("src/demo/download/music.mp3")) { try { Player pl = new Player(in); pl.play(); } catch (JavaLayerException e) { e.printStackTrace(); } } catch (IOException e) { e.printStackTrace(); } } private static void do(String stUr, String Pa) throws IOException { URL ur = new URL(stUr); ReadableByteChannel by = Channels.newChannel(ur.openStream()); FileOutputStream steam = new FileOutputStream(Pa); steam.getChannel().transferFrom(by, 0, Long.MAX_VALUE); steam.close(); by.close(); }}
